<h2 id="官方文档示例">官方文档示例</h2>
<p>注意: 在 jsx 中，应该使用 v-slots 代替 v-slot</p>
<pre><code class="language-javascript">const App = {
  setup() {
    const slots = {
      foo: () => &#x3C;span>B&#x3C;/span>,
    };
    return () => (
      &#x3C;A v-slots={slots}>
        &#x3C;div>A&#x3C;/div>
      &#x3C;/A>
    );
  },
};

// or

const App = {
  setup() {
    const slots = {
      default: () => &#x3C;div>A&#x3C;/div>,
      foo: () => &#x3C;span>B&#x3C;/span>,
    };
    return () => &#x3C;A v-slots={slots} />;
  },
};
</code></pre>
<h2 id="个人示例">个人示例</h2>
<p>xxx.vue</p>
<pre><code class="language-html">&#x3C;van-image
  round
  width="7rem"
  height="7rem"
  fit="cover"
  src="/img/AquaMinato.png"
  style="display: block; margin: 0 auto;"
>
  &#x3C;template v-slot:loading>
    &#x3C;van-loading type="spinner" size="20" />
  &#x3C;/template>
  &#x3C;template v-slot:error>加载失败&#x3C;/template>
&#x3C;/van-image>
</code></pre>
<p>这里我虽然用的是 TSX (TypeScript 的 JSX)，但用法应该和 JSX 是一样的。</p>
<p>xxx.tsx</p>
<pre><code class="language-javascript">setup() {
  const slots = {
    loading: () => &#x3C;van-loading type="spinner" size="20" />,
    error: () => &#x3C;span>加载失败&#x3C;/span>
  }

  return () => (
    &#x3C;div>
      &#x3C;van-image
        round
        width="7rem"
        height="7rem"
        fit="cover"
        src="/img/AquaMinato.png"
        style="display: block; margin: 0 auto;"
        v-slots={slots}
      >
      &#x3C;/van-image>
    &#x3C;/div>
  )
}
</code></pre>
<h2 id="解构插槽-props">解构插槽 Props</h2>
<p>在 template 写法中：</p>
<pre><code class="language-html">&#x3C;router-link
  to="/about"
  custom
  v-slot="{ href, route, navigate, isActive, isExactActive }"
>
  &#x3C;NavLink :active="isActive" :href="href" @click="navigate">
    {{ route.fullPath }}
  &#x3C;/NavLink>
&#x3C;/router-link>
</code></pre>
<p>在 JSX 写法中可以则可以是这样：</p>
<pre><code class="language-javascript">&#x3C;router-link to="/about" custom>
  {{
    default: ({ href, route, navigate, isActive, isExactActive }) => (
      &#x3C;NavLink active={isActive} href={href} click={navigate}>
        {route.fullPath}
      &#x3C;/NavLink>
    ),
  }}
&#x3C;/router-link>
